                var getAllImages = function (items) {
                    var $btns = items;
                    if ($btns.length == 0) {
                        return;
                    }
                    var imageUrls = [];
                    $.each($btns, function (i, btn) {

                var showFancyBox = function (e, images,currentImage) {
                    e.preventDefault();

                    var imageUrls = images;
                    if (!imageUrls || imageUrls.length == 0) {
                        return;
                    }
                    var currentUrl = currentImage.attr('data-value');
                    var before = [];
                    var showArray = [];
                    $.each(imageUrls, function (i, url) {
                        if (url == currentUrl) {
                            showArray.push(url);
                        } else {
                            if (showArray.length > 0) {
                                showArray.push(url);
                            } else {
                                before.push(url);
                            }
                        }
                    });
                    if (before.length > 0) {
                        $.each(before, function (i, url) {
                            showArray.push(url);
                        });
                    }

                    if (showArray && showArray.length > 0) {
                        $.fancybox.open(showArray, {
                            type: 'image'
                        });
                    }
                }

                $(".carousel-inner").on('click','.img_product_screenshot_big_show',function(e){

  <script>
   // footer
    $(".a-feedback").on("click",function(e){
        e.preventDefault();
        $(".div-feedback").show();
        $(".div-feedback-success").hide();
        $(".form-feedback").show();
        //绑定关闭
        $(".div-feedback").find(".close").on("click",function(e){
            e.preventDefault();
            $(".div-feedback").hide();
        });
        //绑定提交
        //防止重复提交机制 ajax提交前加锁 ajax请求返回时解锁
        var lock = false;
        $(".submit-feedback").on("click",function(){
            e.preventDefault();
            var content = $(".form-feedback").find("textarea[name='content']").val();
            if(!content || content.length == 0){
                $(".alert-dismissible").fadeIn();
                setTimeout(function(){
                    $(".alert-dismissible").fadeOut();
                },2000)
                return;
            }
            if(lock){
                return;
            }
            lock = true;
            $.ajax({url:'/api/feedbacks',
                type:'POST',
                data:{item:0,type:"fromIndex",content:content},
                success:function(){
                    $(".form-feedback").resetForm().hide();
                    $(".div-feedback-success").show();
                    $(".button-cancel-feedback").html("我知道了");
                    $(".submit-feedback").hide();
                    lock = false;
                },
                error:function(){
                    //错误处理
                    lock = false;
                }
            });
        });
        //绑定取消
        $(".button-cancel-feedback").on("click",function(e){
            e.preventDefault();
            $(".div-feedback").hide();
        });
    });

    //返回顶部的效果和功能
    $(".fa-chevron-up").parent().hide();
    $(window).scroll(function(){
        if ($(window).scrollTop()>100){
            $(".fa-chevron-up").parent().fadeIn(500);
        }
        else
        {
            $(".fa-chevron-up").parent().fadeOut(500);
        }
    });
    $(".fa-chevron-up").click(function(e){
        e.preventDefault();
        $('body,html').animate({scrollTop:0},500);
        return false;
    });
